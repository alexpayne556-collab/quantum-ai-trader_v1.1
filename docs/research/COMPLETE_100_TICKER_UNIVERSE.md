import yfinance as yf
import pandas as pd

def build_universe_with_filters():
    """
    Build 100-ticker universe with FREE yfinance data
    Apply liquidity filters to maintain quality
    """
    
    # Define all tickers
    all_tickers = {
        'AI_INFRA': ai_infra_leaders + ai_infra_enablers + ai_hardware,
        'ROBOTAXI': robotaxi_pure + robotaxi_enablers,
        'QUANTUM': quantum_hardware + quantum_enablers,
        'GRID_STORAGE': grid_storage_leaders + clean_energy_integration,
        'MEDTECH': medtech_robotics + biotech_ai,
        'AEROSPACE': aerospace_space + advanced_materials,
        'CYBER': cyber_quantum_resistant,
        'BIOTECH': biotech_life_sciences,
        'INDUSTRIAL': industrial_automation,
        'CLOUD': cloud_infrastructure,
        'HEDGES': macro_hedges,
    }
    
    universe = []
    filtered_universe = []
    
    for sector, tickers in all_tickers.items():
        for ticker in tickers:
            try:
                # Get last 60 days of data
                data = yf.download(ticker, period='60d', progress=False)
                
                # Calculate ADV (average daily volume)
                avg_volume = (data['Volume'] * data['Close']).mean()  # Dollar volume
                
                # Calculate spread proxy (high-low)
                spread_pct = ((data['High'] - data['Low']) / data['Close']).mean()
                
                # Get current price
                current_price = data['Close'].iloc[-1]
                
                # Store data
                universe.append({
                    'ticker': ticker,
                    'sector': sector,
                    'current_price': current_price,
                    'adv_dollars': avg_volume,
                    'avg_spread_pct': spread_pct,
                    'liquidity_tier': classify_liquidity(avg_volume),
                })
                
                # Filter: >$5M ADV and <0.5% spread
                if avg_volume > 5_000_000 and spread_pct < 0.005:
                    filtered_universe.append(universe[-1])
                    
            except Exception as e:
                print(f"Skipped {ticker}: {e}")
    
    # Create DataFrame
    universe_df = pd.DataFrame(universe)
    filtered_df = pd.DataFrame(filtered_universe)
    
    print(f"Total tickers analyzed: {len(universe)}")
    print(f"Tickers passing filters (>$5M ADV, <0.5% spread): {len(filtered_df)}")
    print(f"\nSector distribution:")
    print(filtered_df['sector'].value_counts())
    
    return filtered_df

def classify_liquidity(adv_dollars):
    """Classify liquidity tier"""
    if adv_dollars > 50_000_000:
        return 'TIER_A_MEGA'
    elif adv_dollars > 20_000_000:
        return 'TIER_B_LARGE'
    elif adv_dollars > 10_000_000:
        return 'TIER_C_MID'
    elif adv_dollars > 5_000_000:
        return 'TIER_D_SMALL'
    else:
        return 'TIER_E_SKIP'

# RUN:
universe_df = build_universe_with_filters()

def rank_universe_by_breakout_signal(features_df):
    """
    Score all tickers by pre-breakout fingerprint (5-feature ensemble)
    Return top candidates for next 2-4 days
    """
    
    # Pre-breakout score = sum of 5 features
    features_df['breakout_score'] = (
        (features_df['dark_pool_ratio'] > 0.35).astype(int) +
        (features_df['ah_vol_pct'] > 0.30).astype(int) +
        (features_df['options_skew'] < 0.92).astype(int) +
        (abs(features_df['avwap_zscore']) > 2.0).astype(int) +
        (features_df['rsi_7'] > 70).astype(int)
    )
    
    # Rank by score
    ranked = features_df.sort_values('breakout_score', ascending=False)
    
    # Filter: score >= 3 (3/5 features)
    candidates = ranked[ranked['breakout_score'] >= 3].head(20)
    
    print("üöÄ TOP BREAKOUT CANDIDATES (Next 2-4 Days):\n")
    for idx, row in candidates.iterrows():
        print(f"{row['ticker']:6} | {row['sector']:15} | Score: {row['breakout_score']}/5 | "
              f"Price: ${row['price']:.2f}")
    
    return candidates
